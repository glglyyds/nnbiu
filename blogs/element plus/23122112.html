    el-select下拉列表分页显示

    咕噜咕噜 2023/12/21 element plus

    1.新建子组件 select.vue

    <template>
    <el-row>
      <el-col :span="24">
        <el-select v-model="copyValue" filterable clearable placeholder="请选择" @change="updateValue"
          :style="'width:100%;min-width:250px'">
          <el-option v-for="item in dataList" :key="item[valueKey]" :label="item[labelKey]" :value="item[valueKey]">
          </el-option>
          <el-col :span="24">
            <div style="float: right;margin-right:10px;padding-bottom: 10px">
              <el-pagination @current-change="handleCurrentChange" :current-page="page"
                page-size="10" layout="total, prev, pager, next" :total="numTotal">
              </el-pagination>
            </div>
          </el-col>
        </el-select>
      </el-col>
    </el-row>
</template>

    <script>
export default {
    props: {
        handleCurrentChange: {
            type: Function,
            default: function (page) {},
        },
        page: {
            type: Number,
            default: 1,
        },
        numTotal: {
            type: Number,
            default: 0,
        },
        // 绑定值
        value: String,
        // 下拉列表
        dataList: Array,
        // option的label在列表中对应的key
        labelKey: String,
        // option的value在列表中对应的key
        valueKey: String,
    },
    watch: {
      value: {
        handler: function (val) {
          this.copyValue = val;
        },
        deep: true
      }
    },
    methods: {
      // 更新值
      updateValue() {
        this.$emit('update:value', this.copyValue);
      },
    },
    data() {
      return {
        copyValue: this.value
      };
    }
};
</script>

    2.新建父组件 element.vue

    <template>
    <Select :value.sync="copyValue"  :handleCurrentChange="handleCurrentChange" :valueKey="'id'"
     :labelKey="'deviceName'" :page="this.searchObj.page" :numTotal="numTotal" :dataList="entityDataTable">
    </Select>
</template>
    1

    <script>
import Select from '../components/select.vue'
export default {
    components:{Select},
    data(){
       return{
         searchObj:{
            activateStatus: -99,
            deviceCode: "",
            deviceLinkType: -99,
            deviceName: "",
            deviceType: -99,
            ip: "",
            page: 1,
            protocolType: -99
        },
         // 复制值
         copyValue: '',
         // 父实例下拉框的数据
         entityDataTable: [],
         numTotal:''
         }
    },
    mounted(){
        this.getData()
    },
    methods:{
        // 父实例翻页
        handleCurrentChange(val) {
            this.searchObj.page = val;
            this.getData()
        },
        //获取下拉框的数据
        async getData(){
            this.axios.post(this.baseURL + "Device/DeviceList", this.searchObj, this.headers()).then((res) => {
                this.numTotal = res.data.data.total
                this.entityDataTable = res.data.data.deviceList
            });
        }
    }
};
</script>
